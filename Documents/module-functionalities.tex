\section{R K K Jinendra -- Module Owner}

\subsection{Management Staff Records (CRUD)}
\textbf{Purpose:} Maintain records of non-teaching management and administrative manager panel used for access control and operational management.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Add manager panel member profiles 
  \item \textbf{Read:} View manager panel directory, search/filter by name/role/status, and view full profiles.
  \item \textbf{Update:} Edit profile data, deactivate.
  \item \textbf{Delete:} Remove manager panel accounts.
\end{itemize}

\subsubsection*{Typical Fields}
 Full name, NIC, email, phone, address, role, username, created/updated timestamps.

\subsubsection*{Validations and Rules}
\begin{itemize}[leftmargin=*]
  \item Unique identifiers (email).
  \item Role-based access: only authorized management can perform CRUD.
\end{itemize}

\subsection{Teacher Records (CRUD)}
\textbf{Purpose:} Maintain records of teaching management and Teacher panel used for access control and operational management.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Add teacher profiles with qualifications and subject specialties.
  \item \textbf{Read:} List/search teachers; view teacher profile and assigned classes/subjects.
  \item \textbf{Update:} Edit Maintain personal data. 
  \item \textbf{Update:} Assign class to teachers.
  \item \textbf{Delete:} Remove teacher accounts.
\end{itemize}

\subsection{Student Records (CRUD)}
\textbf{Purpose:} Maintain student master data used throughout attendance, results, timetables, and behaviour tracking.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Register admissions and assign student to class/grade.
  \item \textbf{Read:} List students by class/grade, search, and view full student profile.
  \item \textbf{Update:} Update details, class transfer.
  \item \textbf{Delete:} Prefer status changes over hard-delete to preserve historical academic records.
\end{itemize}

\subsubsection*{Validations and Rules}
\begin{itemize}[leftmargin=*]
  \item Unique admission number/student ID.
  \item Student must be linked to a class .

\end{itemize}

\subsection{Parent/Guardian Records (CRUD)}
\textbf{Purpose:} Maintain parent/guardian profiles and link them to one or more students for communication and monitoring.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Add guardian profile and link to student(s).
  \item \textbf{Read:} View guardian directory and linked students.
  \item \textbf{Update:} Update contact details and manage student links.
  \item \textbf{Delete:} Prefer deactivate to keep announcement delivery logs and history.
\end{itemize}

\subsubsection*{Rules}
A student has only one guardians;

\subsection{Student Absence Reason Management (CRUD + Acknowledgement Workflow)}
\textbf{Purpose:} Record reasons for student absences and support acknowledgement by class teachers.

\subsubsection*{CRUD Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Submit absence reason for a student and date/period.
  \item \textbf{Read:} View absence reasons by class teachers.
  \item \textbf{Update:} Edit pending submissions (before acknowledgement) with restrictions.
  \item \textbf{Delete:} Remove absence reasons by acknowledgement.
\end{itemize}

\subsubsection*{Acknowledgement Workflow}
\begin{itemize}[leftmargin=*]
  \item Statuses: \textit{Submitted/Pending} $\rightarrow$ \textit{Acknowledged} or \textit{not}.
  \item An approver can record decision timestamps.
  \item Optionally notify parent/guardian when acknowladged or not.
\end{itemize}

\subsubsection*{Integration}
Linked to student attendance records so that absences can be explained and reported as acknowledged.

\subsection{Student Timetables (CRUD + Publish/Unpublish)}
\textbf{Purpose:} Maintain class timetables.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Build timetable entries per day/period with subject, teacher.
  \item \textbf{Read:} Timetable grid view and printable view.
  \item \textbf{Update:} Edit periods; swap subjects/teachers.
  \item \textbf{Delete:} Remove entries or entire timetable versions.

\end{itemize}

\end{itemize}

\subsection{Relief / Teacher Substitution Management (CRUD)}
\textbf{Purpose:} Assign substitute teachers when a scheduled teacher is absent.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Assign relief teacher for a class, date, and period.
  \item \textbf{Read:} Daily relief schedule (by class and by teacher).
  \item \textbf{Update:} Replace substitute teacher or adjust period.
  \item \textbf{Delete:} Cancel relief assignment.
\end{itemize}

\subsubsection*{Rules}
Prevent timetable conflicts and optionally prefer substitutes qualified for the subject.

\subsection{Exam Timetables (Upload/Update/Delete + Image Handling)}
\textbf{Purpose:} Upload and publish exam schedules, frequently as images/PDFs.

\end{itemize}


\subsection{Teacher Leave Requests (CRUD + Approval Workflow)}
\textbf{Purpose:} Enable teachers to apply for leave and management to approve/reject.

\subsubsection*{CRUD Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Submit leave request (type, date range, reason, attachment).
  \item \textbf{Read:} Teacher views own requests; management views all pending/processed.
  \item \textbf{Update:} Teachers can edit pending requests; approvers add notes.
  \item \textbf{Delete/Cancel:} Cancel pending requests.
\end{itemize}

\subsubsection*{Approval Workflow}
\begin{itemize}[leftmargin=*]
  \item Statuses: \textit{Pending} $\rightarrow$ \textit{Approved} or \textit{Rejected}; optional \textit{Cancelled}.
\end{itemize}

\subsection{Student Behaviour / Remarks (CRUD)}
\textbf{Purpose:} Record behavioural incidents, achievements, counselling notes, and general remarks.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Add remark with category and description.
  \item \textbf{Read:} Student behaviour history and class summaries.
  \item \textbf{Update:} Edit remarks with authorization controls.
  \item \textbf{Delete:} Remove incorrect remarks; preserve audit trail when required.
\end{itemize}

\subsection{Subjects Management (CRUD)}
\textbf{Purpose:} Maintain subject master data used by timetabling, marks, and materials.

\subsubsection*{Functions}
\begin{itemize}[leftmargin=*]
  \item \textbf{Create:} Add subjects with name, grade  type.
  \item \textbf{Read:} View subject catalogue by grade.
  \item \textbf{Delete:} Remove subjects.
\end{itemize}
